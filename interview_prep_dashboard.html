Module 1 — Arrays & Two-Pointers

🟢 Session 1.1 — Basics & Two Sum

P1: Two Sum | Easy O(n) | O(n)

def two_sum(nums, target):
    seen = {}
    for i, v in enumerate(nums):
        if target - v in seen:
            return [seen[target - v], i]
        seen[v] = i
    return []

P2: Best Time to Buy Stock | Easy O(n) | O(1)

def max_profit(prices):
    min_price = prices[0]
    max_profit = 0
    for price in prices[1:]:
        max_profit = max(max_profit, price - min_price)
        min_price = min(min_price, price)
    return max_profit

P3: Merge Sorted Arrays | Easy O(m+n) | O(1)

def merge(nums1, m, nums2, n):
    p1, p2, p = m-1, n-1, m+n-1
    while p1 >= 0 and p2 >= 0:
        nums1[p] = nums1[p1] if nums1[p1] > nums2[p2] else nums2[p2]
        p1 if nums1[p1] > nums2[p2] else (p2:=p2-1); p -= 1
    if p2 >= 0:
        nums1[:p2+1] = nums2[:p2+1]

🟡 Session 1.2 — Sliding Window & Max Subarray

P4: Maximum Subarray (Kadane) | Medium O(n) | O(1)

def max_subarray(nums):
    max_curr = max_global = nums[0]
    for x in nums[1:]:
        max_curr = max(x, max_curr + x)
        max_global = max(max_global, max_curr)
    return max_global

P5: Product of Array Except Self | Medium O(n) | O(n)

def product_except_self(nums):
    n = len(nums)
    left, right = [1]*n, [1]*n
    for i in range(1, n):
        left[i] = left[i-1] * nums[i-1]
    for i in range(n-2, -1, -1):
        right[i] = right[i+1] * nums[i+1]
    return [left[i]*right[i] for i in range(n)]

P6: 3Sum | Medium O(n²) | O(1)

def three_sum(nums):
    nums.sort()
    result = []
    for i in range(len(nums)-2):
        if nums[i] > 0: break
        if i > 0 and nums[i] == nums[i-1]: continue
        left, right = i+1, len(nums)-1
        while left < right:
            total = nums[i] + nums[left] + nums[right]
            if total == 0:
                result.append([nums[i], nums[left], nums[right]])
                while left < right and nums[left] == nums[left+1]: left += 1
                while left < right and nums[right] == nums[right-1]: right -= 1
                left, right = left+1, right-1
            elif total < 0: left += 1
            else: right -= 1
    return result

🟢 Session 2.1 — Basics & Reverse

P1: Reverse Linked List | Easy O(n) | O(1)

def reverse_list(head):
    prev = None
    curr = head
    while curr:
        nxt = curr.next
        curr.next = prev
        prev = curr
        curr = nxt
    return prev

P2: Merge Two Sorted Lists | Easy O(m+n) | O(1)

def merge_two_lists(list1, list2):
    dummy = ListNode(0)
    curr = dummy
    while list1 and list2:
        if list1.val < list2.val:
            curr.next = list1
            list1 = list1.next
        else:
            curr.next = list2
            list2 = list2.next
        curr = curr.next
    curr.next = list1 or list2
    return dummy.next

P3: Find Middle of Linked List | Easy O(n) | O(1)

def middle_node(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow

P4: Detect Cycle (Floyd) | Medium O(n) | O(1)

def has_cycle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow is fast:
            return True
    return False

P5: Remove Nth From End | Medium O(n) | O(1)

def remove_nth_from_end(head, n):
    dummy = ListNode(0)
    dummy.next = head
    first = dummy
    second = dummy
    for _ in range(n+1):
        if not first: return head
        first = first.next
    while first:
        first = first.next
        second = second.next
    second.next = second.next.next
    return dummy.next

P6: Palindrome Linked List | Medium O(n) | O(1)

def is_palindrome(head):
    slow = fast = head
    while fast and fast.next:
        slow, fast = slow.next, fast.next.next
    
    prev = None
    while slow:
        nxt = slow.next
        slow.next = prev
        prev = slow
        slow = nxt
    
    while prev:
        if head.val != prev.val: return False
        head, prev = head.next, prev.next
    return True

🟢 Session 3.1 — Stacks & Valid Parentheses

P1: Valid Parentheses | Easy O(n) | O(n)

def is_valid(s):
    stack = []
    pairs = {')':'(', ']':'[', '}':'{'}
    for ch in s:
        if ch in '([{': 
            stack.append(ch)
        else:
            if not stack or stack[-1] != pairs[ch]: 
                return False
            stack.pop()
    return not stack

P2: Min Stack | Medium O(1) | O(n)

class MinStack:
    def __init__(self):
        self.stack = []
        self.min_stack = []
    
    def push(self, val):
        self.stack.append(val)
        if not self.min_stack or val <= self.min_stack[-1]:
            self.min_stack.append(val)
    
    def pop(self):
        if self.stack[-1] == self.min_stack[-1]:
            self.min_stack.pop()
        self.stack.pop()
    
    def top(self):
        return self.stack[-1]
    
    def getMin(self):
        return self.min_stack[-1]

P3: Evaluate RPN | Medium O(n) | O(n)

def eval_rpn(tokens):
    stack = []
    ops = {'+': lambda a,b: a+b, '-': lambda a,b: a-b,
           '*': lambda a,b: a*b, '/': lambda a,b: int(a/b)}
    for token in tokens:
        if token in ops:
            b = stack.pop()
            a = stack.pop()
            stack.append(ops[token](a, b))
        else:
            stack.append(int(token))
    return stack[0]

🟡 Session 3.2 — Queues & Sliding Window

P4: Sliding Window Maximum | Hard O(n) | O(k)

from collections import deque

def sliding_window_maximum(nums, k):
    deq = deque()
    result = []
    for i, num in enumerate(nums):
        if deq and deq[0][1] < i - k + 1:
            deq.popleft()
        while deq and deq[-1][0] < num:
            deq.pop()
        deq.append((num, i))
        if i >= k - 1:
            result.append(deq[0][0])
    return result

P5: Moving Average from Stream | Easy O(1) | O(size)

from collections import deque

class MovingAverage:
    def __init__(self, size):
        self.size = size
        self.queue = deque()
        self.sum = 0
    
    def next(self, val):
        if len(self.queue) == self.size:
            self.sum -= self.queue.popleft()
        self.queue.append(val)
        self.sum += val
        return self.sum / len(self.queue)
Module 4 — Trees & Binary Search Trees

🟢 Session 4.1 — Traversals & Basics

P1: Inorder Traversal (Iterative) | Easy O(n) | O(h)

def inorder_traversal(root):
    res, stack, curr = [], [], root
    while curr or stack:
        while curr:
            stack.append(curr)
            curr = curr.left
        curr = stack.pop()
        res.append(curr.val)
        curr = curr.right
    return res

P2: Level Order (BFS) | Easy O(n) | O(w)

from collections import deque

def level_order(root):
    if not root: return []
    result, queue = [], deque([root])
    while queue:
        level_size = len(queue)
        level = []
        for _ in range(level_size):
            node = queue.popleft()
            level.append(node.val)
            if node.left: queue.append(node.left)
            if node.right: queue.append(node.right)
        result.append(level)
    return result

P3: Validate BST | Medium O(n) | O(h)

def is_valid_bst(root):
    def validate(node, min_val, max_val):
        if not node: return True
        if node.val <= min_val or node.val >= max_val:
            return False
        return (validate(node.left, min_val, node.val) and
                validate(node.right, node.val, max_val))
    
    return validate(root, float('-inf'), float('inf'))

🟡 Session 4.2 — Advanced Tree Problems

P4: Lowest Common Ancestor | Medium O(n) | O(h)

def lowest_common_ancestor(root, p, q):
    if not root: return None
    if root.val == p.val or root.val == q.val:
        return root
    left = lowest_common_ancestor(root.left, p, q)
    right = lowest_common_ancestor(root.right, p, q)
    if left and right: return root
    return left if left else right

P5: Maximum Path Sum | Hard O(n) | O(h)

def max_path_sum(root):
    max_sum = float('-inf')
    
    def dfs(node):
        nonlocal max_sum
        if not node: return 0
        left = max(0, dfs(node.left))
        right = max(0, dfs(node.right))
        max_sum = max(max_sum, node.val + left + right)
        return node.val + max(left, right)
    
    dfs(root)
    return max_sum

🟡 Session 5.1 — Heap Operations & Kth Problems

P1: Kth Largest Element | Medium O(n log k) | O(k)

import heapq

def find_kth_largest(nums, k):
    # Min-heap of size k, keep k largest
    heap = nums[:k]
    heapq.heapify(heap)
    for num in nums[k:]:
        if num > heap[0]:
            heapq.heapreplace(heap, num)
    return heap[0]

P2: Top K Frequent Elements | Medium O(n log k) | O(n)

import heapq
from collections import Counter

def top_k_frequent(nums, k):
    count = Counter(nums)
    # Min-heap: (frequency, element)
    heap = []
    for num, freq in count.items():
        heapq.heappush(heap, (freq, num))
        if len(heap) > k:
            heapq.heappop(heap)
    return [num for freq, num in heap]

P3: Median of Data Stream | Hard O(log n) | O(n)

import heapq

class MedianFinder:
    def __init__(self):
        self.small = []  # max-heap (negated)
        self.large = []  # min-heap
    
    def addNum(self, num):
        heapq.heappush(self.small, -num)
        if -self.small[0] > self.large[0] if self.large else self.small:
            heapq.heappush(self.large, -heapq.heappop(self.small))
        if len(self.small) < len(self.large):
            heapq.heappush(self.small, -heapq.heappop(self.large))
    
    def findMedian(self):
        if len(self.small) > len(self.large):
            return float(-self.small[0])
        return (-self.small[0] + self.large[0]) / 2

🟡 Session 5.2 — Advanced Heap Techniques

P4: Merge K Sorted Lists | Hard O(n log k) | O(k)

import heapq

def merge_k_lists(lists):
    # Min-heap: (value, list_index, node_index)
    heap = []
    for i, lst in enumerate(lists):
        if lst:
            heapq.heappush(heap, (lst.val, i, 0))
    
    dummy = ListNode(0)
    curr = dummy
    while heap:
        val, list_idx, node_idx = heapq.heappop(heap)
        curr.next = lists[list_idx]
        curr = curr.next
        if curr.next:
            heapq.heappush(heap, (curr.next.val, list_idx, node_idx+1))
    
    return dummy.next

P5: Reorganize String | Medium O(n log n) | O(n)

import heapq
from collections import Counter

def reorganize_string(s):
    count = Counter(s)
    max_freq = max(count.values())
    if max_freq > (len(s)+1)//2:
        return ""
    
    heap = [(-freq, char) for char, freq in count.items()]
    heapq.heapify(heap)
    result = []
    while len(heap) >= 2:
        f1, c1 = heapq.heappop(heap)
        f2, c2 = heapq.heappop(heap)
        result.extend([c1, c2])
        if f1 < -1: heapq.heappush(heap, (f1+1, c1))
        if f2 < -1: heapq.heappush(heap, (f2+1, c2))
    
    if heap:
        result.append(heap[0][1])
    return ''.join(result)
